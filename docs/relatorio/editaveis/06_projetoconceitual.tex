\section{Análise de consumo energético}

\textcolor{red}{Com relação ao consumo energético do produto desenvolvido, será necessário apresentar cálculos de consumo dos diversos componentes utilizados e explicar as decisões de projeto para atender a estas demandas.}


\begin{itemize}
    \item \textcolor{red}{ \textbf{Identificação dos Subsistemas Elétricos:} liste todos os componentes que consumirão energia, como sensores, microcontroladores, atuadores e módulos de armazenamento de dados. Para cada componente, registre a tensão de operação (V), corrente média (mA) e tempo estimado de uso (s).}
    \item \textcolor{red}{ \textbf{Cálculo da Energia Consumida por Componente:} utilize a fórmula: $E = V I t$, onde $E$ é a energia em joules, $V$ é a tensão em volts, $I$ a corrente em ampères e $t$ o tempo em segundos. Realize esse cálculo para cada componente do sistema.}
    \item \textcolor{red}{ \textbf{Estimativa do Consumo Total de Energia:} some as energias de todos os componentes para obter o consumo energético total estimado do sistema por lançamento.}
    \item \textcolor{red}{ \textbf{Escolha da Fonte de Alimentação:} com base no consumo total, converta para watt-hora: 1 Wh = 3600 J. Adicione e justifique uma margem de segurança (pesquisar o que é usual para este tipo de sistema) e selecione uma bateria ou fonte que supra a demanda.}
    \item \textcolor{red}{ \textbf{Planejamento do Circuito de Alimentação:} analise a necessidade de inclusão de reguladores de tensão, proteções contra sobrecorrente, e isolamento para atuadores. Como as oscilações de tensão serão controladas?}
    \item \textcolor{red}{ \textbf{Monitoramento via \textit{software}:} implemente, no \textit{software} embarcado, rotinas para medir a tensão da bateria e registrar dados de tempo de operação. Colete os dados para análise.}
    \item \textcolor{red}{ \textbf{Validação com Testes Reais:} realize testes de campo para comparar o consumo estimado com o real. Ajuste a capacidade da fonte se necessário e refine o software de coleta de dados. Justifique a diferença de resultados (real x teórica), de acordo com a literatura.}
\end{itemize}

\subsection{Detalhamento dos Casos de Teste}

    \subsubsection{CT01 - Validação do Sistema Sensorial e Precisão}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT01 - Leitura e Calibração dos Sensores de Distância (RF03, RF04, RF10, RNF07).
        \item \textbf{Objetivo do caso de teste:} Validar se o sistema reconhece as paredes, adquire os dados corretamente e mede a distância com a precisão mínima exigida.
        \item \textbf{Pré-condições:} Micromouse montado e posicionado em uma célula de teste; monitor serial ou telemetria ativos.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Posicionar o robô a distâncias 6cm, 12cm, 18cm(que corresponde ao tamanho de uma célula) de uma parede.
            \item Registrar o valor reportado pelo sistema para cada sensor.
            \item Verificar se o software identifica a presença de parede quando a distância for inferior ao limite da célula.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O erro de medição deve ser inferior a 1 cm, conforme o requisito RNF07. O sistema deve sinalizar corretamente a existência de obstruções à frente e nas laterais.
    \end{itemize}

    \subsubsection{CT02 - Controle de Trajetória e Estabilidade Elétrica}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT02 - Deslocamento Retilíneo e Controle de Motores (RF01, RF08, RF09, RNF10).
        \item \textbf{Objetivo do caso de teste:} Validar a capacidade de mover-se para frente controlando os motores para evitar colisões sem reinicializações elétricas.
        \item \textbf{Pré-condições:} Bateria carregada; corredor reto livre de obstáculos imprevistos.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Acionar o comando de avanço para percorrer três células consecutivas.
            \item Observar se o robô mantém a trajetória centralizada usando o controle dos motores.
            \item Monitorar se ocorrem quedas de tensão que causem o reset do microcontrolador.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve completar o percurso sem tocar nas paredes laterais e o sistema deve permanecer estável, sem oscilações de tensão que interrompam a execução.
    \end{itemize}

    \subsubsection{CT03 - Precisão de Manobra e Alteração de Direção}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT03 - Rotação e Mudança de Orientação (RF02, RF09).
        \item \textbf{Objetivo do caso de teste:} Garantir que o micromouse realize rotações de até 180° através do controle preciso dos motores.
        \item \textbf{Pré-condições:} Robô parado no centro de uma célula.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Comandar uma rotação de 90° para a esquerda e medir o ângulo final.
            \item Comandar uma rotação de 90° para a direita e medir o ângulo final.
            \item Comandar uma rotação de 180° (meia-volta).
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve realizar as rotações solicitadas com erro angular mínimo, permitindo que ele continue a navegação alinhado ao novo corredor.
    \end{itemize}

    \subsubsection{CT04 - Inteligência e Navegação Autônoma}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT04 - Mapeamento e Busca do Centro (RF05, RF06, RF07, RF14, RNF01).
        \item \textbf{Objetivo do caso de teste:} Validar o funcionamento autônomo na identificação de caminhos, geração do mapa e alcance do objetivo central.
        \item \textbf{Pré-condições:} Labirinto desconhecido pelo robô; algoritmo de navegação (ex: Flood Fill) carregado.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar o robô na célula inicial (0,0) em modo autônomo.
            \item Observar se o robô identifica caminhos livres e mapeia as paredes encontradas.
            \item Verificar se o sistema detecta a chegada ao objetivo central do labirinto.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O micromouse deve encontrar o caminho até o centro de forma autônoma e o mapa gerado internamente deve corresponder à realidade física do labirinto.
    \end{itemize}

    \subsubsection{CT05 - Telemetria e Comunicação sem Fio}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT05 - Validação de Telemetria em Tempo Real (RF16, RF17, RNF04).
        \item \textbf{Objetivo:} Verificar se os dados de navegação são enviados e exibidos na interface web com a latência exigida.
        \item \textbf{Pré-condições:} Micromouse conectado à rede sem fio; sistema web de telemetria ativo e aguardando conexão.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar a movimentação do robô no labirinto.
            \item Observar a atualização dos campos de velocidade e posição na interface web.
            \item Cronometrar o tempo entre um evento físico (ex: detecção de parede) e sua exibição na tela.
        \end{enumerate}
        \item \textbf{Resultado esperado:} Os dados devem ser atualizados a cada 500 ms (RNF05) com latência inferior a 1 segundo (RNF04). A interface deve exibir corretamente a posição X e Y enviada pelo robô.
    \end{itemize}

    \subsubsection{CT06 - Mapeamento e Persistência no Banco de Dados}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT06 - Armazenamento de Tentativas e Mapas (RF13, RF15).
        \item \textbf{Objetivo:} Garantir que os dados da "Tentativa" e as "Posições" percorridas sejam salvos corretamente no banco de dados seguindo o MER.
        \item \textbf{Pré-condições:} Banco de dados operacional; robô ter finalizado uma execução completa.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Executar uma tentativa de travessia do labirinto até o centro.
            \item Acessar o banco de dados e realizar uma consulta na tabela "Tentativa" e "Posicao".
            \item Verificar se os IDs (PK/FK) estão relacionados corretamente entre Micromouse e Tentativa.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O banco de dados deve conter o registro da tentativa com tempo de início/fim e todas as coordenadas (X, Y) percorridas vinculadas ao ID da tentativa realizada.
    \end{itemize}

    \subsubsection{CT07 - Monitoramento Energético e Estabilidade}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT07 - Monitoramento de Tensão e Consumo (RF12, RF18, RF21).
        \item \textbf{Objetivo:} Validar se o sistema monitora a bateria e se mantém estável sob carga (RNF10).
        \item \textbf{Pré-condições:} Robô com bateria em carga parcial (aprox. 50\%).
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar o robô e monitorar o log de tensão da bateria via telemetria.
            \item Simular um esforço maior (aceleração máxima) para observar quedas de tensão (voltage sag).
            \item Verificar se o software registra o consumo estimado em Joules conforme a fórmula $E = V \cdot I \cdot t$ definida no projeto.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O sistema não deve reiniciar durante picos de corrente (RNF10). O nível da bateria deve ser reportado com precisão e o cálculo de consumo energético deve ser salvo ao final da tentativa.
    \end{itemize}

    \subsubsection{CT08 - Conformidade Física e Operação Autônoma}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT08 - Verificação de Dimensões e Autonomia de Início (RNF09, RNF01).
        \item \textbf{Objetivo:} Garantir que o robô respeite os limites físicos do labirinto e inicie sem intervenção humana física.
        \item \textbf{Pré-condições:} Labirinto padrão de 18 cm de largura por célula.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Medir o robô com um paquímetro ou régua.
            \item Posicionar o robô na célula inicial.
            \item Acionar o comando "Iniciar" remotamente via sistema web (RF22).
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve medir menos de 18 cm x 18 cm (RNF09). Após o comando remoto, o robô deve iniciar a exploração de forma totalmente autônoma (RNF01) sem auxílio manual para partida.
    \end{itemize}

    \subsubsection{CT09 - Salvamento e Recuperação do Caminho Percorrido}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT09 - Persistência de Trajetória em Memória (RF11, RF13).
        \item \textbf{Objetivo:} Validar que o caminho completo percorrido pelo robô é salvo corretamente em memória e recuperável após desligamento.
        \item \textbf{Pré-condições:} Robô com memória flash disponível; sistema de arquivos operacional no microcontrolador.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar uma execução do robô no labirinto e deixá-lo navegar por um tempo determinado.
            \item Desligar o robô durante a navegação.
            \item Religar o robô e verificar se consegue acessar o histórico de posições armazenadas.
            \item Comparar o caminho salvo com o caminho esperado baseado nas movimentações registradas.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O sistema deve recuperar o histórico completo de posições e orientações antes do desligamento, permitindo análise posterior da trajetória percorrida.
    \end{itemize}

    \subsubsection{CT10 - Integração de Múltiplos Sensores}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT10 - Validação de Fusão de Dados Sensoriais (RF20, RF04, RF03).
        \item \textbf{Objetivo:} Garantir que todos os sensores de distância funcionam simultaneamente e fornecem dados consistentes durante a navegação.
        \item \textbf{Pré-condições:} Todos os sensores calibrados; célula de teste com configuração conhecida.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Posicionar o robô com diferentes distâncias em relação às paredes (frente, laterais).
            \item Registrar as leituras de todos os sensores simultaneamente.
            \item Verificar a coerência entre os dados (ex: lateral esquerda e direita em corredor simétrico devem ser aproximadamente iguais).
        \end{enumerate}
        \item \textbf{Resultado esperado:} Todos os sensores devem funcionar sem travamentos e fornecer valores consistentes sem discrepâncias anormais entre eles.
    \end{itemize}

    \subsubsection{CT11 - Robustez sob Variações de Iluminação}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT11 - Operação em Diferentes Condições de Luz (RNF06).
        \item \textbf{Objetivo:} Validar que o sistema sensorial do robô funciona corretamente sob variações de iluminação do ambiente.
        \item \textbf{Pré-condições:} Robô em ambiente controlável com possibilidade de ajuste de iluminação; labirinto montado.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Executar uma travessia do labirinto com iluminação ambiente normal.
            \item Repetir a mesma travessia com luz reduzida (50\% de iluminação).
            \item Repetir com luz mínima (aceso apenas com iluminação do robô).
            \item Comparar os resultados das três execuções.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve completar a trajetória com sucesso em todas as condições de iluminação testadas sem perda de orientação ou detecção incorreta de paredes.
    \end{itemize}

    \subsubsection{CT12 - Autonomia Energética Completa}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT12 - Teste de Duração Máxima (RF10, RNF08).
        \item \textbf{Objetivo:} Validar que a bateria fornece autonomia suficiente para completar o desafio do labirinto em menos de 30 minutos conforme RNF08.
        \item \textbf{Pré-condições:} Bateria carregada completamente; labirinto de tamanho competitivo (16x16 células).
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Registrar a tensão inicial da bateria.
            \item Iniciar o robô em modo autônomo para explorar o labirinto completamente.
            \item Cronometrar o tempo total até a conclusão.
            \item Registrar a tensão final da bateria e o consumo energético total calculado.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve completar a exploração e navegação até o centro em menos de 30 minutos, e a bateria deve manter nível superior a 10\% de carga ao final.
    \end{itemize}

    \subsubsection{CT13 - Resistência Estrutural a Impactos}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT13 - Durabilidade Mecânica e Funcionabilidade (RNF11).
        \item \textbf{Objetivo:} Garantir que o robô continua operacional após sofrer múltiplas colisões leves conforme RNF11.
        \item \textbf{Pré-condições:} Robô montado e testado; parede de teste ou célula de colisão preparada.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Realizar 10 colisões leves consecutivas contra uma parede a velocidade reduzida.
            \item Inspecionar visualmente o robô para detectar danos óbvios.
            \item Executar um teste de navegação pós-colisão para validar funcionalidade.
            \item Verificar se sensores e motores ainda respondem corretamente.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve manter todas as funcionalidades após as colisões, sem danos estruturais relevantes ou perda de capacidade sensorial/motora.
    \end{itemize}

    \subsubsection{CT14 - Comunicação Sem Fio em Longo Alcance}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT14 - Conectividade Wireless em Diferentes Distâncias (RNF02, RNF04).
        \item \textbf{Objetivo:} Validar que a comunicação sem fio permanece estável e com latência aceitável em diferentes distâncias entre o robô e o ponto de acesso.
        \item \textbf{Pré-condições:} Sistema de telemetria web ativo; medidor de sinal wireless disponível; labirinto em área com WiFi.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar transmissão de telemetria com o robô a 1 metro do ponto de acesso.
            \item Afastar progressivamente o robô (3m, 5m, 10m) e monitorar a qualidade do sinal.
            \item Registrar a latência e perdas de pacotes em cada distância.
            \item Testar com obstáculos entre o robô e o ponto de acesso (paredes, móveis).
        \end{enumerate}
        \item \textbf{Resultado esperado:} Latência deve permanecer inferior a 1 segundo em até 10 metros de distância com sinal mínimo de -70 dBm. Com obstáculos, deve manter comunicação funcional até 5 metros.
    \end{itemize}

    \subsubsection{CT15 - Tempo de Carregamento da Interface de Telemetria}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT15 - Performance da Página Web (RNF03).
        \item \textbf{Objetivo:} Verificar que a página de telemetria carrega em até 5 segundos conforme RNF03.
        \item \textbf{Pré-condições:} Servidor web em execução; conexão WiFi estável; navegador web atualizado.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Acessar a página de telemetria via URL no navegador.
            \item Medir o tempo desde o carregamento até a exibição completa (primeiros dados visíveis).
            \item Repetir o teste 5 vezes e calcular a média.
            \item Verificar também o carregamento com cache limpo.
        \end{enumerate}
        \item \textbf{Resultado esperado:} Tempo de carregamento completo deve ser inferior a 5 segundos, incluindo renderização de todos os elementos gráficos principais.
    \end{itemize}

    \subsubsection{CT16 - Frequência de Atualização de Telemetria}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT16 - Taxa de Refresh dos Dados (RNF05).
        \item \textbf{Objetivo:} Validar que os dados de telemetria são atualizados a cada 500 ms conforme RNF05.
        \item \textbf{Pré-condições:} Robô em movimento no labirinto; página de telemetria aberta; ferramenta de monitoramento de rede disponível.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar captura de pacotes de rede durante execução do robô.
            \item Medir o intervalo entre atualizações consecutivas de dados de posição.
            \item Registrar a variação no intervalo (jitter).
            \item Verificar se todas as atualizações contêm dados válidos e íntegros.
        \end{enumerate}
        \item \textbf{Resultado esperado:} Intervalo entre atualizações deve ser 500 ms ± 50 ms. Não deve haver perdas de pacotes que causem lacunas maiores que 1 segundo.
    \end{itemize}

    \subsubsection{CT17 - Controle Remoto de Execução via Interface Web}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT17 - Controle de Início e Parada Remota (RF22).
        \item \textbf{Objetivo:} Garantir que o robô pode ser iniciado e parado remotamente pela interface web sem atrasos significativos.
        \item \textbf{Pré-condições:} Robô posicionado na célula inicial; interface web de controle acessível; WiFi conectado.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Clicar no botão "Iniciar" na interface web e cronometrar a resposta do robô.
            \item Deixar o robô navegar por 30 segundos.
            \item Clicar no botão "Parar" e verificar se o robô para em tempo aceitável.
            \item Repetir o ciclo de início e parada 5 vezes.
        \end{enumerate}
        \item \textbf{Resultado esperado:} Tempo de resposta entre clique no botão e ação do robô deve ser inferior a 1 segundo. Parada deve ocorrer em menos de 100 ms após comando.
    \end{itemize}

    \subsubsection{CT18 - Recuperação Após Falha de Sensor}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT18 - Tratamento de Erro de Sensor (RF20, RF04, RF05).
        \item \textbf{Objetivo:} Validar o comportamento do sistema quando um sensor falha ou fornece dados inconsistentes durante a navegação.
        \item \textbf{Pré-condições:} Robô em movimento; possibilidade de bloquear manualmente um sensor; sistema com validação de dados.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Iniciar navegação do robô no labirinto.
            \item Durante a navegação, obstruir físico um dos sensores de distância.
            \item Observar como o sistema reage ao valor anômalo.
            \item Verificar se o robô continua funcionando ou ativa protocolo de segurança.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O sistema deve detectar a leitura anômala, descartar ou suavizar o valor, e continuar navegando de forma segura ou registrar um aviso sem parar completamente.
    \end{itemize}

    \subsubsection{CT19 - Variação de Temperatura Operacional}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT19 - Estabilidade em Diferentes Temperaturas (RNF10).
        \item \textbf{Objetivo:} Validar que o sistema permanece estável e funcional sob variações de temperatura que possam afetar componentes eletrônicos.
        \item \textbf{Pré-condições:} Câmara climática disponível ou ambiente com controle de temperatura; robô totalmente montado.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Executar teste de navegação em temperatura ambiente (20-25°C).
            \item Repetir em ambiente aquecido (30-35°C) simulando competição com iluminação.
            \item Repetir em ambiente mais frio (10-15°C).
            \item Registrar consumo de energia e tensão da bateria em cada condição.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve funcionar sem erros em todas as condições testadas. Consumo de energia pode variar, mas o sistema não deve sofrer reset ou perda de comunicação devido à temperatura.
    \end{itemize}

    \subsubsection{CT20 - Validação Completa do Sistema Integrado}
    \begin{itemize}
        \item \textbf{Nome do caso de teste:} CT20 - Teste de Aceitação Final (Todos os RF/RNF).
        \item \textbf{Objetivo:} Realizar um teste final integrado que valida o sistema completo funcionando de forma holística, cobrindo todos os requisitos funcionais e não-funcionais.
        \item \textbf{Pré-condições:} Sistema totalmente montado e testado; labirinto padrão de competição; interface web operacional; todas as dependências resolvidas.
        \item \textbf{Descrição dos procedimentos:}
        \begin{enumerate}
            \item Carregar bateria completamente.
            \item Posicionar robô na célula inicial (0,0).
            \item Iniciar via comando remoto na interface web.
            \item Monitorar telemetria em tempo real durante toda a execução.
            \item Registrar o tempo de conclusão, consumo energético e qualidade dos dados.
            \item Validar que o mapa foi armazenado corretamente no banco de dados.
        \end{enumerate}
        \item \textbf{Resultado esperado:} O robô deve completar uma execução bem-sucedida do labirinto com todos os requisitos atendidos: navegação autônoma, coleta de dados correta, comunicação sem perdas significativas, e conclusão em menos de 30 minutos.
    \end{itemize}
